PROGRAM Z.
TYPE-POOLS VRM.
DATA:    PARAM    TYPE VRM_ID,
         ITVALUES TYPE VRM_VALUES,
         WA       LIKE LINE OF ITVALUES.

PARAMETERS: P_NAME(10) AS LISTBOX VISIBLE LENGTH 30.

AT SELECTION-SCREEN OUTPUT.
  PARAM = 'P_NAME'.
  WA-KEY = '1'. WA-TEXT = 'JOHN'.
  APPEND WA TO ITVALUES.
  WA-KEY = '2'. WA-TEXT = 'PETER'.
  APPEND WA TO ITVALUES.
  CALL FUNCTION 'VRM_SET_VALUES'
    EXPORTING
      ID     = PARAM
      VALUES = ITVALUES.

START-OF-SELECTION.
  WRITE P_NAME.
