REPORT Z_CL_CLASS_02_DEMOPRG.
START-OF-SELECTION.
PARAMETERS: PCARRID TYPE S_CARRID.

DATA OBJ TYPE REF TO ZCL_EVENT_CLASS01.

DATA: WA_FLIGHT TYPE ZLT_SFLIGHT,
      IT_FLIGHT TYPE ZTT_SFLIGHT_RT.

CREATE OBJECT OBJ.
SET HANDLER OBJ->NO_FLIGHT_HANDLER FOR OBJ.



CALL METHOD OBJ->GET_FLIGHT
  EXPORTING
    IM_CARRID = PCARRID
  IMPORTING
    EX_FLIGHT = IT_FLIGHT
    .

CALL METHOD OBJ->DISPLAY_FLIGHT
  EXPORTING
    IM_FLIGHT = IT_FLIGHT
    .
