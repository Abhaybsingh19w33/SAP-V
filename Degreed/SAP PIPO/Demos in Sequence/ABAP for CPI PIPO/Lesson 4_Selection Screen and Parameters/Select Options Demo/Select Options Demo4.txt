REPORT Z.
DATA: BEGIN OF SCARR,
     CARRID TYPE SCARR-CARRID,
     CARRNAME TYPE SCARR-CARRNAME,
     CURRCODE TYPE SCARR-CURRCODE,
     URL TYPE SCARR-URL,
END OF SCARR.
DATA ITSCARR TYPE TABLE OF SCARR.
DATA WA TYPE SCARR.
SELECT-OPTIONS SCARRID FOR SCARR-CARRID.

WRITE SCARRID-LOW.
WRITE SCARRID-HIGH.
WRITE SCARRID-SIGN.
WRITE SCARRID-OPTION.

SELECT CARRID CARRNAME  CURRCODE FROM SCARR
  INTO TABLE ITSCARR
  WHERE CARRID IN SCARRID.

LOOP AT ITSCARR INTO WA.
  WRITE :/ WA-CARRID , WA-CARRNAME , WA-CURRCODE, WA-URL.
ENDLOOP.
