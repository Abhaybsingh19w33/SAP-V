REPORT  Z.
CLASS NUMBERS DEFINITION.
  PUBLIC SECTION.
    METHODS SETNUM.
    METHODS ADDNUM.
    METHODS DISPNUM.

  PRIVATE SECTION.
    DATA: NUM1 TYPE I,
          NUM2 TYPE I,
          RES  TYPE I.
ENDCLASS.

CLASS NUMBERS IMPLEMENTATION.
  METHOD SETNUM.
    NUM1 = 10.
    NUM2 = 20.
  ENDMETHOD.

  METHOD ADDNUM.
    RES = NUM1 + NUM2.
  ENDMETHOD.

  METHOD DISPNUM.
    WRITE:/ 'Res is:' , RES.
  ENDMETHOD.
ENDCLASS.

START-OF-SELECTION.
  DATA NUM TYPE REF TO NUMBERS.
  CREATE OBJECT NUM.
  NUM->SETNUM( ).
  NUM->ADDNUM( ).
  NUM->DISPNUM( ).
